55gf_engine_t* gf_engine_create(const char* title, int nogui) {
56 gf_engine_t* engine = malloc(sizeof(*engine));
57 memset(engine, 0, sizeof(*engine));
58 engine->log = stderr;
59 if(nogui) {
60 gf_log_function(engine, "No GUI mode", "");
61 engine->client = NULL;
62 } else {
63 gf_log_function(engine, "GUI mode", "");
64 engine->client = gf_client_create(engine, title);
65 if(engine->client == NULL) {
66 gf_log_function(engine, "Failed to create client interface", "");
67 gf_engine_destroy(engine);
68 return NULL;
69 }
70 gf_log_function(engine, "Switching to graphical console", "");
71 }
72 engine->server = gf_server_create(engine);
73
74 engine->base = gf_resource_create(engine, "base.pak");
75 if(!nogui) {
76 engine->client->draw->font = gf_font_create_file(engine->client->draw, "base:/font/default.bdf");
77 }
78 return engine;
79}
