54void gf_gui_destroy_id(gf_gui_t* gui, gf_gui_id_t id) {
55 int i;
56 int ind = hmgeti(gui->area, id);
57 gf_gui_component_t* c;
58 if(ind == -1) return;
59
60 c = &gui->area[ind];
61 switch(c->type) {
62 case GF_GUI_BUTTON: {
63 if(c->u.button.text != NULL) free(c->u.button.text);
64 c->u.button.text = NULL;
65 }
66 case GF_GUI_WINDOW: {
67 if(c->u.window.title != NULL) free(c->u.window.title);
68 c->u.window.title = NULL;
69 }
70 }
71 for(i = 0; i < hmlen(gui->area); i++) {
72 if(gui->area[i].parent == id) {
73 gf_gui_destroy_id(gui, gui->area[i].key);
74 i--;
75 }
76 }
77
78 if(c->prop != NULL) {
79 shfree(c->prop);
80 }
81
82 hmdel(gui->area, id);
