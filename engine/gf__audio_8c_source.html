25void gf_audio_callback(ma_device* dev, void* output, const void* input, ma_uint32 frame) {
26 ma_uint32 i;
27 gf_audio_t* audio = dev->pUserData;
28 ma_int16* out = (ma_int16*)output;
29 float* tmp = malloc(sizeof(*tmp) * frame * 2);
30 int unlocked = 0;
31
32 for(i = 0; i < frame; i++) {
33 tmp[2 * i + 0] = 0;
34 tmp[2 * i + 1] = 0;
35 }
36
37 ma_mutex_lock(audio->mutex);
38 for(i = 0; i < (ma_uint32)hmlen(audio->decoder); i++) {
39 if(audio->decoder[i].used == 1 && audio->decoder[i].decoder != NULL) {
40 ma_uint64 readframe;
41 int j;
42 ma_int16* r = malloc(sizeof(*r) * frame * 2);
43 ma_decoder_read_pcm_frames(audio->decoder[i].decoder, r, frame, &readframe);
44 for(j = 0; j < readframe; j++) {
45 tmp[2 * j + 0] += (double)r[2 * j + 0] / 32768.0;
46 tmp[2 * j + 1] += (double)r[2 * j + 1] / 32768.0;
47 }
48 free(r);
49 if(frame > readframe) {
50 ma_mutex_unlock(audio->mutex);
51 unlocked = 1;
52 gf_audio_decoder_destroy(&audio->decoder[i]);
53 }
54 } else if(audio->decoder[i].used == 1 && audio->decoder[i].xm != NULL) {
55 int j;
56 int gotframe;
57 float* r = malloc(sizeof(*r) * frame * 2);
58 jar_xm_generate_samples(audio->decoder[i].xm, r, frame);
59 gotframe = jar_xm_get_last(audio->decoder[i].xm) != -1 ? jar_xm_get_last(audio->decoder[i].xm) : (int)frame;
60 for(j = 0; j < gotframe; j++) {
61 tmp[2 * j + 0] += (double)r[2 * j + 0];
62 tmp[2 * j + 1] += (double)r[2 * j + 1];
63 }
64 free(r);
65 if(jar_xm_get_loop_count(audio->decoder[i].xm) > audio->decoder[i].internal) {
66 ma_mutex_unlock(audio->mutex);
67 unlocked = 1;
68 gf_audio_decoder_destroy(&audio->decoder[i]);
69 }
70 } else if(audio->decoder[i].used == 1 && audio->decoder[i].mod != NULL) {
71 int j;
72 int gotframe;
73 ma_int16* r = malloc(sizeof(*r) * frame * 2);
74 jar_mod_fillbuffer(audio->decoder[i].mod, r, frame, NULL);
75 gotframe = audio->decoder[i].mod->last != -1 ? audio->decoder[i].mod->last : (int)frame;
76 for(j = 0; j < gotframe; j++) {
77 tmp[2 * j + 0] += (double)r[2 * j + 0] / 32768.0;
78 tmp[2 * j + 1] += (double)r[2 * j + 1] / 32768.0;
79 }
80 free(r);
81 if(audio->decoder[i].mod->loopcount > audio->decoder[i].internal) {
82 ma_mutex_unlock(audio->mutex);
83 unlocked = 1;
84 gf_audio_decoder_destroy(&audio->decoder[i]);
85 }
86 }
87 }
88 if(!unlocked) ma_mutex_unlock(audio->mutex);
89
90 for(i = 0; i < frame; i++) {
91 out[2 * i + 0] = tmp[2 * i + 0] * 32768;
92 out[2 * i + 1] = tmp[2 * i + 1] * 32768;
93 }
94 free(tmp);
95}

97gf_audio_id_t gf_audio_load(gf_audio_t* audio, const void* data, size_t size) {
98 gf_audio_decoder_t decoder;
99 int xm_cond;
100 int mod_cond;
101 int ind;
102
103 ma_mutex_lock(audio->mutex);
104
105 decoder.used = 0;
106 decoder.audio = audio;
107 decoder.decoder = NULL;
108 decoder.xm = NULL;
109 decoder.mod = NULL;
110 decoder.decoder_config = ma_decoder_config_init(audio->device_config.playback.format, audio->device_config.playback.channels, audio->device_config.sampleRate);
111 decoder.key = 0;
112 do {
113 ind = hmgeti(audio->decoder, decoder.key);
114 if(ind != -1) {
115 decoder.key++;
116 }
117 } while(ind != -1);
118
119 xm_cond = size > 37 && memcmp(data, "Extended Module: ", 17) == 0 && ((char*)data)[37] == 0x1a;
120 mod_cond = size > 1080;
121
122 if(mod_cond) {
123 int j;
124 int mod_sig_cond = 0;
125 for(j = 0; j < sizeof(gf_audio_mod_sig) / sizeof(gf_audio_mod_sig[0]); j++) {
126 mod_sig_cond = mod_sig_cond || (memcmp((char*)data + 1080, gf_audio_mod_sig[j], 4) == 0);
127 }
128 mod_cond = mod_cond && mod_sig_cond;
129 }
130
131 if(xm_cond) {
132 if(jar_xm_create_context_safe(&decoder.xm, data, size, audio->device_config.sampleRate) == 0) {
133 /* In XM loader .internal is used to store old loopcount */
134 decoder.internal = jar_xm_get_loop_count(decoder.xm);
135 decoder.used = -1;
136 jar_xm_set_max_loop_count(decoder.xm, 1);
137 hmputs(audio->decoder, decoder);
138 ma_mutex_unlock(audio->mutex);
139 return decoder.key;
140 }
141 decoder.xm = NULL;
142 } else if(mod_cond) {
143 decoder.mod = malloc(sizeof(*decoder.mod));
144 jar_mod_init(decoder.mod);
145 decoder.mod->modfile = malloc(size);
146 decoder.mod->modfilesize = size;
147 memcpy(decoder.mod->modfile, data, size);
148 jar_mod_setcfg(decoder.mod, audio->device_config.sampleRate, 16, 1, 1, 0);
149 if(jar_mod_load(decoder.mod, (void*)decoder.mod->modfile, size)) {
150 /* In MOD loader .internal is used to store old loopcount */
151 decoder.internal = decoder.mod->loopcount;
152 decoder.used = -1;
153 hmputs(audio->decoder, decoder);
154 ma_mutex_unlock(audio->mutex);
155 return decoder.key;
156 }
157 free(decoder.mod->modfile);
158 free(decoder.mod);
159 decoder.mod = NULL;
160 }
161 decoder.decoder = malloc(sizeof(*decoder.decoder));
162 if(ma_decoder_init_memory(data, size, &decoder.decoder_config, decoder.decoder) == MA_SUCCESS) {
163 decoder.used = -1;
164 hmputs(audio->decoder, decoder);
165 ma_mutex_unlock(audio->mutex);
166 return decoder.key;
167 }
168 free(decoder.decoder);
169 decoder.decoder = NULL;
170 ma_mutex_unlock(audio->mutex);
171 return -1;
172}
