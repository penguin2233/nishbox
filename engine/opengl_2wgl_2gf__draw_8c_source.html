31LRESULT CALLBACK gf_draw_platform_proc(HWND hWnd, UINT msg, WPARAM wp, LPARAM lp) {
32 PAINTSTRUCT ps;
33 RECT rect;
34 gf_draw_t* draw = (gf_draw_t*)GetWindowLongPtr(hWnd, GWLP_USERDATA);
35 switch(msg) {
36 case WM_PAINT:
37 BeginPaint(hWnd, &ps);
38 EndPaint(hWnd, &ps);
39 break;
40 case WM_SIZE:
41 if(draw->platform != NULL) {
42 GetClientRect(hWnd, &rect);
43 draw->x = rect.left;
44 draw->y = rect.top;
45 draw->width = rect.right - rect.left;
46 draw->height = rect.bottom - rect.top;
47 wglMakeCurrent(draw->platform->dc, draw->platform->glrc);
48 gf_draw_reshape(draw);
49 }
50 break;
51 case WM_MOUSEMOVE:
52 if(draw->input != NULL) {
53 POINTS p = MAKEPOINTS(lp);
54 draw->input->mouse_x = p.x;
55 draw->input->mouse_y = p.y;
56 }
57 break;
58 case WM_LBUTTONDOWN:
59 case WM_LBUTTONUP:
60 if(draw->input != NULL) {
61 if(msg == WM_LBUTTONDOWN) {
62 SetCapture(hWnd);
63 draw->input->mouse_flag |= GF_INPUT_MOUSE_LEFT_MASK;
64 } else {
65 draw->input->mouse_flag ^= GF_INPUT_MOUSE_LEFT_MASK;
66 ReleaseCapture();
67 }
68 }
69 break;
70 case WM_MBUTTONDOWN:
71 case WM_MBUTTONUP:
72 if(draw->input != NULL) {
73 if(msg == WM_MBUTTONDOWN) {
74 SetCapture(hWnd);
75 draw->input->mouse_flag |= GF_INPUT_MOUSE_MIDDLE_MASK;
76 } else {
77 draw->input->mouse_flag ^= GF_INPUT_MOUSE_MIDDLE_MASK;
78 ReleaseCapture();
79 }
80 }
81 break;
82 case WM_RBUTTONDOWN:
83 case WM_RBUTTONUP:
84 if(draw->input != NULL) {
85 if(msg == WM_RBUTTONDOWN) {
86 SetCapture(hWnd);
87 draw->input->mouse_flag |= GF_INPUT_MOUSE_RIGHT_MASK;
88 } else {
89 draw->input->mouse_flag ^= GF_INPUT_MOUSE_RIGHT_MASK;
90 ReleaseCapture();
91 }
92 }
93 break;
94 case WM_CLOSE:
95 draw->close = 1;
96 break;
97 case WM_DESTROY:
98 PostQuitMessage(0);
99 break;
100 default:
101 return DefWindowProc(hWnd, msg, wp, lp);
102 }
103 return 0;
104}

106int gf_draw_platform_has_extension(gf_draw_t* draw, const char* query) {
107 const char* ext = NULL;
108 const char* ptr;
109 const int len = strlen(query);
110 PFNWGLGETEXTENSIONSSTRINGARB proc;
111
112 wglMakeCurrent(draw->platform->dc, draw->platform->glrc);
113
114 proc = (PFNWGLGETEXTENSIONSSTRINGARB)wglGetProcAddress("wglGetExtensionsStringARB");
115
116 if(proc != NULL) {
117 ext = proc(draw->platform->dc);

124int gf_draw_platform_step(gf_draw_t* draw) {
125 MSG msg;
126 int ret = 0;
127 wglMakeCurrent(draw->platform->dc, draw->platform->glrc);
128 while(PeekMessage(&msg, draw->platform->window, 0, 0, PM_NOREMOVE)) {
129 if(GetMessage(&msg, draw->platform->window, 0, 0)) {
130 TranslateMessage(&msg);
131 DispatchMessage(&msg);
132 } else {
133 ret = 1;
134 break;
135 }
136 }
137 if(ret == 0) {
138 gf_draw_driver_before(draw);
139 gf_draw_frame(draw);
140 gf_draw_driver_after(draw);
141
142 SwapBuffers(draw->platform->dc);
143 }
144 return ret;
145}
